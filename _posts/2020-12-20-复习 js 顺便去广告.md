(function() {
    'use strict';
    var urlHost = window.location.host;
    var times = 0, timer;
    switch (urlHost){
        case 'segmentfault.com': // https://segmentfault.com/a/1190000038527882
            timer = setInterval(function(){
                times ++;
                if(times == 50){
                    clearInterval(timer);
                }
                //console.log("asd");

                    document.getElementsByClassName('col-12')[1].remove();
                }catch(e){}finally{}
            },10);
            break;
        case 'www.w3cschool.cn': // https://www.w3cschool.cn/electronmanual/9xri1qlc.html
            timer = setInterval(function(){
                times ++;
                if(times == 50){
                    clearInterval(timer);
                }
                document.getElementsByClassName('widget-main')[0].remove();
                document.getElementById('topbanner').remove();
                document.getElementById('rfbanner').remove();
            }, 100);
            break;
        case 'www.tutorialspoint.com': // https://www.tutorialspoint.com/electron/how_electron_works.htm
            timer = setInterval(function () {
                times ++;
                if (times == 100) {
                    clearInterval(timer);
                }
                //console.log("asd");
                document.querySelectorAll('[id^=ad-container]').forEach(function (ele) {
                    var parent = ele.parentNode.parentNode;
                    parent.parentNode.removeChild(parent);
                })
            }, 100);
            break;
        case (urlHost.match(/.*zhihu.com/) || {}).input: // https://www.zhihu.com/question/293069047   https://zhuanlan.zhihu.com/p/34139993
            timer = setInterval(function () {

                }catch{}finally{}

                times ++;
                if (times == 200) {
                    clearInterval(timer);
                }
                document.getElementsByClassName('AdblockBanner')[0].remove();
            }, 10);
            break;
        case 'www.geeksforgeeks.org': // https://www.geeksforgeeks.org/python-lambda/?ref=lbp

                times ++;
                if (times == 200) {
